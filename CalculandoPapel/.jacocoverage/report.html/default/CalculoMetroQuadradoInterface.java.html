    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        CalculoMetroQuadrado cm = new CalculoMetroQuadrado();
        
        cm.setLargura(Integer.parseInt(txtLargura.getText()));
        cm.setAlturaMilimetro(Integer.parseInt(txtAlturaMilimetro.getText()));
        cm.setEsqueleto(Integer.parseInt(txtEsqueleto.getText()));
        cm.setAlturaEsquelto(Integer.parseInt(txtAlturaEsqueleto.getText()));
        cm.setQtd(Integer.parseInt(txtEsqueleto.getText()));
        
        JOptionPane.showMessageDialog(this, cm.Total());
               
        txtLargura.setText("");
        txtAlturaMilimetro.setText("");
        txtEsqueleto.setText("");
        txtAlturaEsqueleto.setText("");
        txtQtd1.setText("");
        
        campoValor2.setText(String.valueOf(cm.Total()));
       
        
        txtLargura.requestFocus(); 
    }//GEN-LAST:event_jButton1ActionPerformed
