    private void txtgerarValorActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_txtgerarValorActionPerformed
        CalculoCimento c = new CalculoCimento();
        c.setAreia(Float.parseFloat(txtareia.getText()));
        c.setPedra(Float.parseFloat(txtpedra.getText()));
        c.setCimento(Float.parseFloat(txtcimento.getText()));
        c.setM3(Float.parseFloat(txtmetroCubico.getText()));
        
        JOptionPane.showMessageDialog(this, c.calculoAreia());
        JOptionPane.showMessageDialog(this, c.calculoPedra());
        JOptionPane.showMessageDialog(this, c.calculoCimento());
        
        txtareia.setText("");
        txtpedra.setText("");
        txtcimento.setText("");
        txtmetroCubico.setText("");
        
        txtgravar2.setText(String.valueOf(c.calculoAreia()));
        txtgravar1.setText(String.valueOf(c.calculoPedra()));
        txtgravar.setText(String.valueOf(c.calculoCimento()));

        // TODO add your handling code here:
    }//GEN-LAST:event_txtgerarValorActionPerformed
